namespace = cx_hive_world

## Origin Start
country_event = {
	id = cx_hive_world.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_origin = origin_hive_cx_hive_world
	}

	immediate = {
		random_owned_planet ={
			limit = { 
				is_capital = yes
			}
			validate_planet_buildings_and_districts = yes
			set_planet_entity = {
				entity = "infected_planet_01_stage_3_entity"
			}
		}
	}
}

# Stage 1 Complete
planet_event = {
	id = cx_hive_world.2
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		remove_modifier = cx_hive_stage_1
		set_planet_entity = {
			entity = infected_planet_01_stage_2_entity
		}
		add_modifier = {
			modifier = cx_hive_stage_2
			days = 4320
		}
		planet_event = { id = cx_hive_world.3 days = 4320 } # 12 Years (20 Total)
	}
}

# Stage 2 Complete
planet_event = {
	id = cx_hive_world.3
	hide_window = yes

	is_triggered_only = yes
	
	immediate = {
		remove_modifier = cx_hive_stage_1
		remove_modifier = cx_hive_stage_2

		change_pc = pc_hive
		set_planet_entity = {
				entity = infected_planet_01_stage_3_entity
		}
	}
}

# Modified to allow player hives to self purge
citizenship_purge = {
	new_pop_weight = 0.0

	is_purge = yes

	potential = {
		exists = from
		NOT = { from = { is_machine_empire = yes } }
		OR = {
			NOT = { has_trait = trait_mechanical }
			from = { has_policy_flag = robots_outlawed }
			AND = {
				from = { is_hive_empire = yes }
				is_robotic_species = yes
			}
		}
	}

	allow = {
		hidden_trigger = { exists = from }
		if = {
			limit = { is_lonely_hive_mind_pop = yes }
			hidden_trigger = { always = yes }
		}
		else = {
			custom_tooltip = {
				fail_text = is_broken_shackles_origin
				if = {
					limit = {
						OR = {
							is_robotic = no
							AND = {
								is_robotic = yes
								NOT = {
									from = { has_policy_flag = robots_outlawed }
								}
							}
						}
					}
					from = {
						NOT = {
							has_origin = origin_broken_shackles
						}
					}
				}
			}
			custom_tooltip = {
				fail_text = same_species_disallowed
				NOT = { is_same_species = from }
			}
			custom_tooltip = {
				fail_text = "presapients_not_purged_citizenship"
				OR = {
					is_sapient = yes
					from = {
						NOT = {
							has_policy_flag = pre_sapients_protect
							has_policy_flag = pre_sapients_allow
							has_policy_flag = pre_sapients_eat
						}
					}
				}
			}
			custom_tooltip = {
				fail_text = symbiote_disallowed
				NOT = { is_symbiote_of_from = yes }
			}
			if = {
				limit = {
					NOR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
					}
				}
				custom_tooltip = {
					fail_text = PURGING_NOT_ALLOWED
					from = { allows_purge = yes }
				}
			}
			if = {
				limit = {
					from = {
						is_hive_empire = yes
						is_ai = yes # ANZ Modified
					}
					has_trait = trait_hive_mind
				}
				custom_tooltip = {
					fail_text = hive_cant_purge_hive
					has_genesis_preserve_trait = no
				}
			}
			custom_tooltip = {
				fail_text = yes_service_yes_citizenship
				if = {
					limit = {
						from = { has_valid_civic = civic_citizen_service }
					}
					NOT = { has_military_service_type = { type = military_service_full country = from } }
				}
			}
			custom_tooltip = {
				fail_text = selective_kinship_must_citizen
				if = {
					limit = {
						from = { has_valid_civic = civic_selective_kinship }
					}
					NOT = { is_same_species_class = from }
				}
			}
		}
	}
	ai_will_do = {
		factor = 20
		modifier = {
			factor = 0
			has_trait = trait_machine_unit
			from = {
				is_machine_empire = yes
			}
		}
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = no
				is_hive_empire = no
			}
		}
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOR = {
					has_ai_personality_behaviour = purger
					AND = {
						has_ai_personality_behaviour = displacer
						if = {
							limit = {
								has_origin = origin_necrophage
							}
							is_hive_empire = yes
						}
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				is_machine_empire = yes
				has_government = gov_machine_industrial
			}
		}
		modifier = {
			factor = 0
			from = {
				is_hive_empire = yes
				NOT = { has_valid_civic = civic_hive_devouring_swarm }
				NOT = { has_origin = origin_necrophage }
				OR = {
					has_valid_civic = civic_hive_one_mind
					has_valid_civic = civic_hive_subsumed_will
					root = { has_trait = trait_hive_mind }
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				has_origin = origin_syncretic_evolution
			}
		}
		modifier = {
			factor = 10
			from = {
				is_ai = yes
				has_ascension_perk = ap_become_the_crisis
			}
		}
		modifier = {
			add = 1000
			not = { has_species_flag = syncretic_species@from }
			is_robotic = no
			from = {
				has_origin = origin_necrophage
				is_xenophobe = yes
			}
		}
	}
}

# Can Purge Robots/Machine Units as Machine Empire
citizenship_purge_machine = {
	new_pop_weight = 0.0
	
	is_purge = yes

	potential = {
		exists = from
		from = { is_machine_empire = yes }
		if = {
			limit = {
				from = { is_ai = yes }
			}
			NOR = {
				has_trait = trait_machine_unit
				has_trait = trait_mechanical
				AND = {
					has_trait = trait_cybernetic
					has_trait = trait_zombie
					from = {
						has_ascension_perk = ap_mechromancy
					}
				}
			}
			NAND = {
				is_lonely_hive_mind_pop = no
				from = { has_valid_civic = civic_machine_servitor }
			}
		}
	}

	allow = {
		hidden_trigger = { exists = from }

		# Block purging Main Species (unless you are player)
		if = {
			limit = {
				from = { is_ai = yes }
			}
			custom_tooltip = {
				fail_text = same_species_disallowed
				NOT = { is_same_species = from }
			}
		}		

		# Servitors cannot purge organics
		if = { limit = { from = { has_civic = civic_machine_servitor } }
			custom_tooltip = {
				fail_text = SERVITORS_CANNOT_PURGE_ORGANICS
				OR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
				}
			}
		}
		
		custom_tooltip = {
			fail_text = "presapients_not_purged_citizenship"
			OR = {
				is_sapient = yes
				from = {
					NOR = {
						has_policy_flag = pre_sapients_protect
						has_policy_flag = pre_sapients_allow
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 0
	}
}
